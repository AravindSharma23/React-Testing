import { screen,render } from "@testing-library/react";
import Application from "./Application";

describe('Application',()=>{
    test('RENDERS CORRECTLY',()=>{
        render(<Application/>)
        // const textElement = screen.getByRole("textbox"); // if single textbox present means use this
        // expect(textElement).toBeInTheDocument();
        const pageHeading = screen.getByRole("heading",
            {
                level:1,  // h1 tag
                name:"Job Application Form"
            });
        expect(pageHeading).toBeInTheDocument();
        
        const sectionText = screen.getByRole("heading",
            {
                level :2, // h2 tag
                name : "Section 1"
            }
        )
        expect(sectionText).toBeInTheDocument();

        const txtAreaElement = screen.getByRole("textbox",{name:"Name"});// name value should be Label content or button content or aria-label content
        expect(txtAreaElement).toBeInTheDocument();
        
        const bioElement = screen.getByRole("textbox",{name:"Bio"});
        expect(bioElement).toBeInTheDocument();

        const jobLocationElement = screen.getByRole("combobox");
        expect(jobLocationElement).toBeInTheDocument();

  
    })
})
